\begin{equation} \begin{split} \vec{u}_A \cdot \vec{n} &= 0 \,,\\ \varphi_{A} &= 0\,. \end{split}\tag{4.2.8} \end{equation}

\begin{equation} \begin{split} u_{1A} &= 0 \,,\\ \varphi_{A} &= 0\,. \end{split}\tag{4.2.9} \end{equation}

\begin{equation} \begin{split} u_{2A} &= 0 \,,\\ \varphi_{A} &= 0\,. \end{split}\tag{4.2.10} \end{equation}
