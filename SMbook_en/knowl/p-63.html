\begin{align*} \regulardiff{\chi_1}{X_1} = 3\,X_1^2\,,\amp\regulardiff{\chi_1}{X_2} = 0\,,\\ \regulardiff{\chi_2}{X_1} = 0\,,\amp\regulardiff{\chi_2}{X_2} = 3\,X_2^2 \end{align*}

\begin{equation*} \func{\mat{F}}{\vec{X}} = \left[\begin{array}{cc} \regulardiff{\chi_1}{X_1} \amp \regulardiff{\chi_1}{X_2} \\ \regulardiff{\chi_2}{X_1} \amp \regulardiff{\chi_2}{X_2} \end{array}\right] = \left[\begin{array}{cc} 3\,X_1^2 \amp 0 \\ 0 \amp 3\,X_2^2 \end{array}\right] \end{equation*}
