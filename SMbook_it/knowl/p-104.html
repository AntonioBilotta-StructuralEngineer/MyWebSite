\begin{equation*} \mat{U} = \left[\begin{array}{cc} 1.3416 \amp 0.4472 \\ 0.4472 \amp 0.8944 \end{array}\right]\,,\quad \mat{R} = \left[\begin{array}{cc} 0.8944 \amp 0.4472 \\ -0.4472 \amp 0.8944 \end{array}\right]\,. \end{equation*}
